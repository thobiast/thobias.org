#!/bin/sed -f
#
# Converte arquivo texto em html
# Última atualização Ter Mai  6 10:58:48 BRT 2003

s/&/\&amp;/g
s/</\&lt;/g
s/>/\&gt;/g
s/"/\&quot;/g
s/'/\&#039;/g

s,\(http\|ftp\)://[\~A-Za-z0-9\.-/_-]\+[A-Za-z0-9],<a href=&>&</a>,g
s/\([A-Za-z0-9_\.-]\+@\)\([A-Za-z0-9_]\+\.\)\+\([A-Za-z]\)\+/<a href=mailto:&>&<\/a>/g

1 i \
<html>\
<head>\
<title>\
txt2html.sed\
</title>\
</head>\
<body>\
<pre> 

$ a \
<\pre>\
<\body>\
<\html>
